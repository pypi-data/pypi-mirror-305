version: '1.1'
var1:
  description: A suffix variable
  default:
    - val1
    - val2
dyn{{ suffix }}:
  type: dynamic
  dynamic:
    type: variable
    variable: _.var1
    propertyerror: false
  var:
    description: A dynamic variable
var2:
  description: A variable calculated
  default:
    type: variable
    variable: _.dynval1.var
----
