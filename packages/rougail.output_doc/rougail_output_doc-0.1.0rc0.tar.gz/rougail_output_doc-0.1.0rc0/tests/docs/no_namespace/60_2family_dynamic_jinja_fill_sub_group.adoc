== dictionaries/rougail/00-base.yml

[,yaml]
----
version: '1.1'
var1:  # a suffix variable
  - val1
  - val2
dyn{{ suffix }}:
  description: a dynamic family
  type: dynamic
  dynamic:
    type: variable
    variable: _.var1
  family:  # a family
    var:  # with a variable
var2:
  description: a second variable
  default:
    type: jinja
    jinja: |
      {{ _.dynval1.family.var }}
    description: the value of var
----
