    async def _update_node_width(self, value: int):
        int_value = int(value)
        await self.graph.update_node_style("2-3", {
            "width": int_value
        })

    async def _update_node_height(self, value: str):
        int_value = int(value)
        await self.graph.update_node_style("2-3", {
            "height": int_value
        })
